Question 1

Write a function to implement a binary search algorithm.

O(log n) — see the full discussion in the Coding level. Key points for an “advanced” variant: handle edge cases, off-by-one carefully, and know lower/upper-bound variants.

function binarySearch(arr, target) {
  let lo = 0, hi = arr.length - 1;
  while (lo <= hi) {
    const mid = lo + ((hi - lo) >> 1);     // avoids overflow
    if (arr[mid] === target) return mid;
    if (arr[mid] < target)    lo = mid + 1;
    else                       hi = mid - 1;
  }
  return -1;
}

// Lower bound — first index where arr[i] >= target
function lowerBound(arr, target) {
  let lo = 0, hi = arr.length;
  while (lo < hi) {
    const mid = (lo + hi) >> 1;
    if (arr[mid] < target) lo = mid + 1;
    else                    hi = mid;
  }
  return lo;
}

Prefer the half-open [lo, hi) form for bounds — cleaner for range queries.

Question 2

How do you create a memoized version of a function?

Advanced memoization: LRU cache, object keys via WeakMap, async support.

function memoize(fn, { max = Infinity, keyFn = JSON.stringify } = {}) {
  const cache = new Map();
  return function (...args) {
    const key = keyFn(args);
    if (cache.has(key)) {
      const val = cache.get(key);
      cache.delete(key); cache.set(key, val);      // bump to most-recent
      return val;
    }
    const result = fn.apply(this, args);
    cache.set(key, result);
    if (cache.size > max) cache.delete(cache.keys().next().value);
    return result;
  };
}

// Async memoize — cache the promise so in-flight calls dedupe
function memoizeAsync(fn) {
  const cache = new Map();
  return (...args) => {
    const key = JSON.stringify(args);
    if (!cache.has(key)) {
      cache.set(key, fn(...args).catch(e => { cache.delete(key); throw e; }));
    }
    return cache.get(key);
  };
}

The async version de-duplicates concurrent calls — critical for rate-limited APIs.

Question 3

Implement a function to perform merge sort on an array.

See Coding Q42 for the full implementation. Advanced considerations:

// Iterative bottom-up — avoids stack depth for huge arrays
function mergeSortIter(arr) {
  const n = arr.length;
  const buf = new Array(n);
  for (let width = 1; width < n; width *= 2) {
    for (let i = 0; i < n; i += 2 * width) {
      mergeInto(arr, buf, i, Math.min(i + width, n), Math.min(i + 2*width, n));
    }
    [arr, buf] = [buf, arr];         // swap buffers
  }
  return arr;
}
function mergeInto(src, dst, lo, mid, hi) {
  let i = lo, j = mid, k = lo;
  while (i < mid && j < hi) dst[k++] = src[i] <= src[j] ? src[i++] : src[j++];
  while (i < mid) dst[k++] = src[i++];
  while (j < hi)  dst[k++] = src[j++];
}

Stable O(n log n). Pre-allocating the buffer avoids repeated allocation. Natural merge sort exploits existing sorted runs for better constants on real data.

Question 4

Write a function to find the longest increasing subsequence in an array.

O(n log n) via patience sorting — see Coding Q43 for explanation. Reconstruction of the sequence itself (not just length):

function longestIncreasingSubseq(nums) {
  if (!nums.length) return [];
  const n = nums.length;
  const parent = new Array(n).fill(-1);
  const tailIdx = [];            // tailIdx[i] = index in nums of smallest tail of length i+1

  for (let i = 0; i < n; i++) {
    let lo = 0, hi = tailIdx.length;
    while (lo < hi) {
      const mid = (lo + hi) >> 1;
      if (nums[tailIdx[mid]] < nums[i]) lo = mid + 1;
      else                                hi = mid;
    }
    if (lo > 0) parent[i] = tailIdx[lo - 1];
    tailIdx[lo] = i;
  }

  // Reconstruct
  const result = [];
  let k = tailIdx[tailIdx.length - 1];
  while (k !== -1) { result.push(nums[k]); k = parent[k]; }
  return result.reverse();
}

longestIncreasingSubseq([10,9,2,5,3,7,101,18]); // [2,3,7,101]
Question 5

Create a function to serialize and deserialize a binary tree.

Pre-order traversal with null markers. See Coding Q46 for implementation. Advanced variant — BFS (level-order) serialization for wider trees:

function serializeBFS(root) {
  if (!root) return "[]";
  const q = [root], out = [];
  while (q.length) {
    const node = q.shift();
    if (!node) { out.push("null"); continue; }
    out.push(node.val);
    q.push(node.left);
    q.push(node.right);
  }
  return "[" + out.join(",") + "]";
}

function deserializeBFS(data) {
  if (data === "[]") return null;
  const tokens = data.slice(1, -1).split(",");
  const root = { val: +tokens[0], left: null, right: null };
  const q = [root];
  let i = 1;
  while (q.length && i < tokens.length) {
    const node = q.shift();
    if (tokens[i] !== "null") {
      node.left = { val: +tokens[i], left: null, right: null };
      q.push(node.left);
    }
    i++;
    if (i < tokens.length && tokens[i] !== "null") {
      node.right = { val: +tokens[i], left: null, right: null };
      q.push(node.right);
    }
    i++;
  }
  return root;
}

BFS uses less bytes for balanced trees; DFS wins for sparse/skewed trees.

Question 6

Implement a function to check if two strings are anagrams.

See Coding Q7 for the core algorithm. Advanced variants:

// Unicode-safe (graphemes)
function isAnagramUnicode(a, b) {
  const normalize = (s) => [...s.normalize("NFKC")].sort().join("");
  return normalize(a) === normalize(b);
}

// Accent-insensitive
function isAnagramLoose(a, b) {
  const clean = (s) => s.toLowerCase()
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, "")    // strip combining marks
    .replace(/[^\p{L}\p{N}]/gu, "");    // keep letters + numbers only
  const sorted = (s) => [...clean(s)].sort().join("");
  return sorted(a) === sorted(b);
}

isAnagramLoose("Café", "face");   // true

String.prototype.normalize resolves combining-character issues ("é" vs "e + ́"). Essential when input comes from users typing accented text.

Question 7

Write a function to find the k-th largest element in an array.

See Coding Q47. Production-grade version using a real min-heap:

class MinHeap {
  constructor() { this.h = []; }
  size() { return this.h.length; }
  peek() { return this.h[0]; }

  push(v) {
    this.h.push(v);
    let i = this.h.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.h[p] <= this.h[i]) break;
      [this.h[p], this.h[i]] = [this.h[i], this.h[p]];
      i = p;
    }
  }
  pop() {
    const top = this.h[0];
    const last = this.h.pop();
    if (this.h.length) {
      this.h[0] = last;
      let i = 0, n = this.h.length;
      for (;;) {
        const l = 2*i+1, r = 2*i+2;
        let m = i;
        if (l < n && this.h[l] < this.h[m]) m = l;
        if (r < n && this.h[r] < this.h[m]) m = r;
        if (m === i) break;
        [this.h[m], this.h[i]] = [this.h[i], this.h[m]];
        i = m;
      }
    }
    return top;
  }
}

function kthLargest(nums, k) {
  const heap = new MinHeap();
  for (const n of nums) {
    heap.push(n);
    if (heap.size() > k) heap.pop();
  }
  return heap.peek();
}

O(n log k). Better than full sort when k << n.

Question 8

Create a function to find the intersection of two arrays.

See Coding Q11. Advanced variants handle multiple arrays and preserve frequencies.

// Intersection of k arrays — O(total size)
function intersectAll(arrays) {
  if (!arrays.length) return [];
  return arrays.reduce((acc, arr) => {
    const set = new Set(arr);
    return acc.filter(x => set.has(x));
  });
}

// Multi-set intersection (preserve min frequency)
function intersectMultiset(a, b) {
  const countA = new Map();
  for (const x of a) countA.set(x, (countA.get(x) || 0) + 1);
  const out = [];
  for (const x of b) {
    const c = countA.get(x);
    if (c > 0) { out.push(x); countA.set(x, c - 1); }
  }
  return out;
}

intersectAll([[1,2,3],[2,3,4],[3,4,5]]);  // [3]
intersectMultiset([1,2,2,1], [2,2]);       // [2,2]
Question 9

Implement a debounce function.

See Coding Q8. Advanced: leading+trailing edge, cancelation, immediate flush.

function debounce(fn, wait, { leading = false, trailing = true } = {}) {
  let timer = null;
  let lastArgs, lastThis;
  let result;

  function invoke() { result = fn.apply(lastThis, lastArgs); }

  function debounced(...args) {
    lastArgs = args;
    lastThis = this;
    const callNow = leading && !timer;
    clearTimeout(timer);
    timer = setTimeout(() => {
      timer = null;
      if (trailing && !callNow) invoke();
    }, wait);
    if (callNow) invoke();
    return result;
  }

  debounced.cancel = () => { clearTimeout(timer); timer = null; };
  debounced.flush  = () => { if (timer) { clearTimeout(timer); timer = null; invoke(); } };

  return debounced;
}

Matches Lodash's API. The cancel/flush methods are crucial for components that unmount — flush any pending call, or cancel it, before teardown.

Question 10

Write a function to perform deep cloning of an object with circular references.

See Advanced Q11 for the approach. The full version with prototypes preserved and cycle handling:

function deepClone(value, seen = new WeakMap()) {
  if (value === null || typeof value !== "object") return value;
  if (value instanceof Date)   return new Date(value);
  if (value instanceof RegExp) return new RegExp(value.source, value.flags);
  if (value instanceof Map) {
    if (seen.has(value)) return seen.get(value);
    const out = new Map(); seen.set(value, out);
    for (const [k, v] of value) out.set(deepClone(k, seen), deepClone(v, seen));
    return out;
  }
  if (value instanceof Set) {
    if (seen.has(value)) return seen.get(value);
    const out = new Set(); seen.set(value, out);
    for (const v of value) out.add(deepClone(v, seen));
    return out;
  }
  if (seen.has(value)) return seen.get(value);
  const out = Array.isArray(value) ? [] : Object.create(Object.getPrototypeOf(value));
  seen.set(value, out);
  for (const key of Reflect.ownKeys(value)) {
    out[key] = deepClone(value[key], seen);
  }
  return out;
}

Or use the built-in structuredClone which handles all these cases and is substantially faster.

Question 11

Create a function to flatten a deeply nested array.

See Coding Q3. Production: iterative stack-based to avoid recursion limits.

function flatten(arr) {
  const stack = [...arr];
  const out = [];
  while (stack.length) {
    const next = stack.pop();
    if (Array.isArray(next)) stack.push(...next);
    else out.push(next);
  }
  return out.reverse();
}

// Depth-limited
function flattenDepth(arr, depth = 1) {
  return arr.reduce((acc, val) =>
    acc.concat(Array.isArray(val) && depth > 0 ? flattenDepth(val, depth - 1) : val),
    []);
}

// Native — most efficient
arr.flat(Infinity);

arr.flat(Infinity) handles any depth. The iterative form is defensive against pathologically deep arrays that would blow the JS stack with recursion.

Question 12

Implement a function to find the maximum sum subarray (Kadane’s Algorithm).

Kadane's algorithm — see Coding Q41. Variant: 2D version for matrix problems.

function maxSubarraySum2D(matrix) {
  const rows = matrix.length, cols = matrix[0].length;
  let best = -Infinity;

  for (let top = 0; top < rows; top++) {
    const col_sum = new Array(cols).fill(0);
    for (let bottom = top; bottom < rows; bottom++) {
      for (let c = 0; c < cols; c++) col_sum[c] += matrix[bottom][c];
      // 1D Kadane on col_sum
      let cur = col_sum[0], max = col_sum[0];
      for (let c = 1; c < cols; c++) {
        cur = Math.max(col_sum[c], cur + col_sum[c]);
        max = Math.max(max, cur);
      }
      best = Math.max(best, max);
    }
  }
  return best;
}

O(rows² · cols). The outer loops pick a row range; the inner 1D Kadane finds the best column range.

Question 13

Write a function to solve the N-Queens problem.

See Coding Q70 for N-Queens. Advanced: bit-mask version is ~4× faster and O(1) per conflict check.

function solveNQueens(n) {
  let count = 0;
  function place(row, cols, d1, d2) {
    if (row === n) { count++; return; }
    let available = ~(cols | d1 | d2) & ((1 << n) - 1);
    while (available) {
      const bit = available & -available;          // lowest set bit
      available ^= bit;
      place(row + 1, cols | bit, (d1 | bit) << 1, (d2 | bit) >> 1);
    }
  }
  place(0, 0, 0, 0);
  return count;
}

solveNQueens(8); // 92

Each bit in cols, d1, d2 tracks an attacked file/diagonal. Shifting the diagonal masks each row naturally propagates the threat. 32-bit ints limit n ≤ 31; use BigInt for larger.

Question 14

Create a function to generate all subsets of an array.

See Coding Q39. Advanced: generate on demand via generator (avoids materializing 2ⁿ).

function* subsets(arr) {
  const n = arr.length;
  for (let mask = 0; mask < (1 << n); mask++) {
    const sub = [];
    for (let i = 0; i < n; i++) {
      if (mask & (1 << i)) sub.push(arr[i]);
    }
    yield sub;
  }
}

// Backtracking version — easier to adapt for pruning
function* subsetsBacktrack(arr, start = 0, cur = []) {
  yield [...cur];
  for (let i = start; i < arr.length; i++) {
    cur.push(arr[i]);
    yield* subsetsBacktrack(arr, i + 1, cur);
    cur.pop();
  }
}

// Take first 10
let n = 0;
for (const s of subsets([1,2,3,4,5])) { if (++n > 10) break; console.log(s); }

Use a generator for early termination — don't compute all 2ⁿ if you only need a few.

Question 15

Implement a function to check if a binary tree is height-balanced.

See Coding Q67 — post-order DFS with early exit.

function isBalanced(root) {
  const check = (node) => {
    if (!node) return 0;
    const l = check(node.left);
    if (l === -1) return -1;
    const r = check(node.right);
    if (r === -1) return -1;
    if (Math.abs(l - r) > 1) return -1;
    return 1 + Math.max(l, r);
  };
  return check(root) !== -1;
}

O(n) single pass. Returning -1 as the “unbalanced” sentinel lets us bail out early without exceptions.

Question 16

Write a function to find the longest palindromic substring.

See Coding Q50. Advanced: Manacher's algorithm in O(n).

function longestPalindrome(s) {
  // Transform to handle even/odd uniformly: "abc" -> "^#a#b#c#$"
  const t = "^#" + s.split("").join("#") + "#$";
  const p = new Array(t.length).fill(0);
  let C = 0, R = 0;

  for (let i = 1; i < t.length - 1; i++) {
    const mirror = 2 * C - i;
    if (i < R) p[i] = Math.min(R - i, p[mirror]);
    while (t[i + p[i] + 1] === t[i - p[i] - 1]) p[i]++;
    if (i + p[i] > R) { C = i; R = i + p[i]; }
  }

  let maxLen = 0, center = 0;
  for (let i = 1; i < p.length - 1; i++) {
    if (p[i] > maxLen) { maxLen = p[i]; center = i; }
  }
  return s.slice((center - maxLen) >> 1, ((center + maxLen) >> 1));
}

Manacher's is O(n). The O(n²) expand-around-center version is simpler and usually fast enough.

function numIslandsUF(grid) {
  const rows = grid.length;
  if (!rows) return 0;
  const cols = grid[0].length;

  const parent = new Array(rows * cols).fill(-1);
  let count = 0;

  const find = (x) => { while (parent[x] !== x) { parent[x] = parent[parent[x]]; x = parent[x]; } return x; };
  const union = (a, b) => { const ra = find(a), rb = find(b); if (ra !== rb) { parent[ra] = rb; count--; } };

  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      if (grid[r][c] === "1") {
        const id = r * cols + c;
        parent[id] = id;
        count++;
        if (r > 0 && grid[r-1][c] === "1") union(id, (r-1)*cols + c);
        if (c > 0 && grid[r][c-1] === "1") union(id, r*cols + (c-1));
      }
    }
  }
  return count;
}

function maxPathSum(root) {
  let best = -Infinity;

    if (!node) return 0;
    const l = Math.max(0, dfs(node.left));
    const r = Math.max(0, dfs(node.right));

  };

function throttle(fn, wait, { leading = true, trailing = true } = {}) {

  return function throttled(...args) {

    }
  };
}

function morrisInorder(root) {
  const out = [];

  while (cur) {
    if (!cur.left) { out.push(cur.val); cur = cur.right; }
    else {
      let pred = cur.left;
      while (pred.right && pred.right !== cur) pred = pred.right;
      if (!pred.right) { pred.right = cur; cur = cur.left; }       // thread

function quicksort(arr, lo = 0, hi = arr.length - 1) {
  if (lo < hi) {
    const p = partition(arr, lo, hi);
    quicksort(arr, lo, p - 1);
    quicksort(arr, p + 1, hi);
  }
  return arr;

function partition(arr, lo, hi) {
  // Median-of-three pivot to avoid O(n²) on sorted input
  const mid = (lo + hi) >> 1;
  if (arr[lo] > arr[mid]) [arr[lo], arr[mid]] = [arr[mid], arr[lo]];
  if (arr[lo] > arr[hi])  [arr[lo], arr[hi]]  = [arr[hi], arr[lo]];
  if (arr[mid] > arr[hi]) [arr[mid], arr[hi]] = [arr[hi], arr[mid]];
  [arr[mid], arr[hi - 1]] = [arr[hi - 1], arr[mid]];

  const pivot = arr[hi - 1];
  let i = lo;
  for (let j = lo; j < hi - 1; j++) {
    if (arr[j] < pivot) {
      [arr[i], arr[j]] = [arr[j], arr[i]];
      i++;
    }
  }
  [arr[i], arr[hi - 1]] = [arr[hi - 1], arr[i]];
  return i;
}

Not stable. Worst-case O(n²) on pathological input; median-of-three pivot selection usually prevents this. JS engines use TimSort internally via Array.prototype.sort, which is stable O(n log n).

}

    fuel += pop();
    stops++;

}

Greedy: drive as far as current fuel allows, then refuel at whichever past station had the most gas. The max-heap gives us the best previous option in O(log n).

  if (exp < 0) { base = 1 / base; exp = -exp; }

    if (exp & 1) result *= base;
    base *= base;
    exp >>>= 1;

}

function topologicalSort(numNodes, edges) {
  const adj = Array.from({length: numNodes}, () => []);
  const indegree = new Array(numNodes).fill(0);
  for (const [u, v] of edges) {
    adj[u].push(v);
    indegree[v]++;
  }

  for (let i = 0; i < numNodes; i++) if (indegree[i] === 0) queue.push(i);

  while (queue.length) {

    order.push(node);
    for (const neighbor of adj[node]) {
      if (--indegree[neighbor] === 0) queue.push(neighbor);
    }
  }

}

function tsp(dist) {
  const n = dist.length;
  const dp = Array.from({length: 1 << n}, () => new Array(n).fill(Infinity));
  dp[1][0] = 0;                      // start at city 0

  for (let mask = 1; mask < (1 << n); mask++) {
    for (let u = 0; u < n; u++) {
      if (!(mask & (1 << u))) continue;
      for (let v = 0; v < n; v++) {
        if (mask & (1 << v)) continue;
        const newMask = mask | (1 << v);
        if (dp[mask][u] + dist[u][v] < dp[newMask][v]) {
          dp[newMask][v] = dp[mask][u] + dist[u][v];
        }
      }
    }

  let best = Infinity;
  for (let u = 0; u < n; u++) best = Math.min(best, dp[full][u] + dist[u][0]);
  return best;
}

Cycle decomposition: count elements displaced by sorting.

function minSwapsToSort(arr) {
  const indexed = arr.map((v, i) => [v, i])
                     .sort((a, b) => a[0] - b[0]);
  const visited = new Array(arr.length).fill(false);
  let swaps = 0;
  for (let i = 0; i < arr.length; i++) {
    if (visited[i] || indexed[i][1] === i) continue;

    let j = i;

  }

}

Question 78

Implement a function to solve the minimum spanning tree problem.

Kruskal's algorithm with Union-Find.

function mst(n, edges) {
  // edges: [[u, v, weight], ...]
  edges.sort((a, b) => a[2] - b[2]);

  const parent = Array.from({length: n}, (_, i) => i);
  const find = (x) => { while (parent[x] !== x) { parent[x] = parent[parent[x]]; x = parent[x]; } return x; };

  let total = 0, count = 0;
  const result = [];
  for (const [u, v, w] of edges) {
    const ru = find(u), rv = find(v);
    if (ru !== rv) {
      parent[ru] = rv;
      total += w;
      result.push([u, v, w]);
      if (++count === n - 1) break;
    }
  }
  return { weight: total, edges: result };
}

O(E log E). Prim's algorithm (with a priority queue) is better when the graph is dense — O(E log V).

Question 79

Write a function to find the shortest path in a weighted graph.

Dijkstra's algorithm with a min-heap. Assumes non-negative weights.

function dijkstra(graph, start) {
  // graph[u] = [[v, weight], ...]
  const n = graph.length;
  const dist = new Array(n).fill(Infinity);
  dist[start] = 0;
  // Simple heap (sorted array for brevity)
  const heap = [[0, start]];

  while (heap.length) {
    const [d, u] = heap.shift();
    if (d > dist[u]) continue;
    for (const [v, w] of graph[u]) {
      const nd = d + w;
      if (nd < dist[v]) {
        dist[v] = nd;
        // insert in order
        let pos = 0;
        while (pos < heap.length && heap[pos][0] < nd) pos++;
        heap.splice(pos, 0, [nd, v]);
      }
    }
  }
  return dist;
}

For negative weights use Bellman-Ford (O(VE)). For dense graphs, Floyd-Warshall (O(V³)) gives all-pairs shortest paths.

Question 80

Create a function to solve the maximum flow problem.

Ford-Fulkerson (Edmonds-Karp variant uses BFS).

function maxFlow(graph, source, sink) {
  const n = graph.length;
  const capacity = Array.from({length: n}, () => new Array(n).fill(0));
  for (let u = 0; u < n; u++) {
    for (const [v, w] of graph[u]) capacity[u][v] = w;
  }

  const bfs = (parent) => {
    parent.fill(-1);
    parent[source] = -2;
    const q = [[source, Infinity]];
    while (q.length) {
      const [u, flow] = q.shift();
      for (let v = 0; v < n; v++) {
        if (parent[v] === -1 && capacity[u][v] > 0) {
          parent[v] = u;
          const newFlow = Math.min(flow, capacity[u][v]);
          if (v === sink) return newFlow;
          q.push([v, newFlow]);
        }
      }
    }
    return 0;
  };

  let total = 0;
  const parent = new Array(n);
  let flow;
  while ((flow = bfs(parent)) !== 0) {
    total += flow;
    let v = sink;
    while (v !== source) {
      const u = parent[v];
      capacity[u][v] -= flow;
      capacity[v][u] += flow;
      v = u;
    }
  }
  return total;
}

Edmonds-Karp runs in O(VE²).

Write a function to solve the stable marriage problem.

Gale-Shapley algorithm.

function stableMarriage(menPrefs, womenPrefs) {
  const n = menPrefs.length;
  // Rank map for women: rank[w][m] = index in w's preference list
  const rank = Array.from({length: n}, () => new Array(n));
  for (let w = 0; w < n; w++) {
    womenPrefs[w].forEach((m, i) => rank[w][m] = i);
  }

  const husbandOf = new Array(n).fill(-1);
  const nextProposal = new Array(n).fill(0);    // next index in man's pref list
  const freeMen = [...Array(n).keys()];

  while (freeMen.length) {
    const m = freeMen.shift();
    const w = menPrefs[m][nextProposal[m]++];
    if (husbandOf[w] === -1) {
      husbandOf[w] = m;
    } else if (rank[w][m] < rank[w][husbandOf[w]]) {
      freeMen.push(husbandOf[w]);
      husbandOf[w] = m;
    } else {
      freeMen.push(m);
    }
  }
  return husbandOf.map((m, w) => [m, w]);     // [husband, wife] pairs
}

Question 85

Write a function to solve the job scheduling problem.

Weighted interval scheduling with DP.

function jobScheduling(startTime, endTime, profit) {
  const n = startTime.length;
  const jobs = Array.from({length: n}, (_, i) => [startTime[i], endTime[i], profit[i]]);
  jobs.sort((a, b) => a[1] - b[1]);                     // by end time

  const dp = new Array(n).fill(0);

  // Binary search for latest non-overlapping job
  const find = (i) => {
    let lo = 0, hi = i - 1;
    while (lo <= hi) {
      const mid = (lo + hi) >> 1;
      if (jobs[mid][1] <= jobs[i][0]) lo = mid + 1;
      else                             hi = mid - 1;
    }
    return hi;
  };

  dp[0] = jobs[0][2];
  for (let i = 1; i < n; i++) {
    const prev = find(i);
    const take = jobs[i][2] + (prev === -1 ? 0 : dp[prev]);
    const skip = dp[i - 1];
    dp[i] = Math.max(take, skip);
  }
  return dp[n - 1];
}

O(n log n). Sort by end time; for each job, binary search the latest compatible one, then choose take/skip.

rob([2, 7, 9, 3, 1]); // 12  (2 + 9 + 1)

O(n) time, O(1) space. The “House Robber” problem. Variant: circular array (first and last are adjacent) — run twice, skipping first house or last house, take the max.

Question 94

Write a function to find the minimum number of jumps to reach the end of an array.

Greedy — track farthest reachable position and next-jump boundary.

function jump(nums) {
  let jumps = 0, curEnd = 0, farthest = 0;
  for (let i = 0; i < nums.length - 1; i++) {
    farthest = Math.max(farthest, i + nums[i]);
    if (i === curEnd) {
      jumps++;
      curEnd = farthest;
    }
  }
  return jumps;
}

jump([2,3,1,1,4]); // 2 (2 -> 3 -> 4)

O(n). When we reach the end of the current jump's range, we commit to jumping and extend to the farthest reachable from within that range.

Question 95

Create a function to find the maximum length of a subarray with positive product.

Track max and min product lengths; sign flips swap them.

function maxLenPositiveProduct(nums) {
  let pos = 0, neg = 0, best = 0;
  for (const n of nums) {
    if (n === 0) {
      pos = 0; neg = 0;
    } else if (n > 0) {
      pos++;
      neg = neg ? neg + 1 : 0;
    } else {
      const newPos = neg ? neg + 1 : 0;
      const newNeg = pos + 1;
      pos = newPos; neg = newNeg;
    }
    best = Math.max(best, pos);
  }
  return best;
}

maxLenPositiveProduct([1,-2,-3,4]); // 4

Zero resets both. Positive extends pos; negative swaps the roles (new pos from prior neg, new neg from prior pos).

Question 96

Implement a function to find the minimum cost to climb stairs.

DP variant of Fibonacci with costs.

function minCostClimbingStairs(cost) {
  let prev2 = 0, prev1 = 0;
  for (let i = 2; i <= cost.length; i++) {
    const cur = Math.min(prev1 + cost[i-1], prev2 + cost[i-2]);
    prev2 = prev1;
    prev1 = cur;
  }
  return prev1;
}

minCostClimbingStairs([10, 15, 20]); // 15

Start for free from step 0 or 1, reach top = position n. Each step cost accumulates when leaving that step.

Question 97

Write a function to find the number of ways to paint a fence with n posts using k colors.

DP: track two states — same color as previous, or different.

function numWaysToPaintFence(n, k) {
  if (n === 0) return 0;
  if (n === 1) return k;
  let same = k;                       // both posts same color
  let diff = k * (k - 1);             // different colors
  for (let i = 3; i <= n; i++) {
    const newSame = diff;             // new same requires previous to differ
    const newDiff = (same + diff) * (k - 1);
    same = newSame;
    diff = newDiff;
  }
  return same + diff;
}

numWaysToPaintFence(3, 2); // 6

Constraint: no three adjacent posts can be the same color. The recurrence captures this by only allowing same-same when the prior two differed.

Question 98

Create a function to find the maximum number of non-overlapping intervals.

Greedy sort by end time — classic interval scheduling.

function maxNonOverlapping(intervals) {
  intervals.sort((a, b) => a[1] - b[1]);
  let count = 0, lastEnd = -Infinity;
  for (const [start, end] of intervals) {
    if (start >= lastEnd) {
      count++;
      lastEnd = end;
    }
  }
  return count;
}

maxNonOverlapping([[1,2],[2,3],[3,4],[1,3]]); // 3

Equivalent to “erase overlap intervals”: intervals.length - maxNonOverlapping. Sorting by end time is provably optimal for this greedy choice.

Question 99

Implement a function to find the length of the shortest transformation sequence from a start word to an end word.

Word Ladder — BFS on a word graph.

function ladderLength(beginWord, endWord, wordList) {
  const words = new Set(wordList);
  if (!words.has(endWord)) return 0;

  let queue = new Set([beginWord]);
  const visited = new Set([beginWord]);
  let steps = 1;

  while (queue.size) {
    const next = new Set();
    for (const word of queue) {
      if (word === endWord) return steps;
      for (let i = 0; i < word.length; i++) {
        for (let c = 97; c <= 122; c++) {
          const ch = String.fromCharCode(c);
          if (ch === word[i]) continue;
          const candidate = word.slice(0, i) + ch + word.slice(i + 1);
          if (words.has(candidate) && !visited.has(candidate)) {
            visited.add(candidate);
            next.add(candidate);
          }
        }
      }
    }
    queue = next;
    steps++;
  }
  return 0;
}

Bidirectional BFS (start from both ends, expand the smaller frontier) roughly halves the work.

Question 100

Write a function to find the longest common substring of two strings.

Longest common substring (contiguous, unlike LCS) — see Advanced Coding Q87.

function longestCommonSubstring(a, b) {
  const m = a.length, n = b.length;
  const dp = Array.from({length: m+1}, () => new Array(n+1).fill(0));
  let best = 0, end = 0;

  for (let i = 1; i <= m; i++) {
    for (let j = 1; j <= n; j++) {
      if (a[i-1] === b[j-1]) {
        dp[i][j] = dp[i-1][j-1] + 1;
        if (dp[i][j] > best) { best = dp[i][j]; end = i; }
      }
    }
  }
  return a.slice(end - best, end);
}

longestCommonSubstring("ABABC", "BABCA"); // "BABC"

O(m·n) time, reducible to O(min(m, n)) space with rolling row. Suffix-tree approaches achieve O(m+n) but are complex.
