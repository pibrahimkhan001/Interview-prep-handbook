Question 1

Write a function to reverse a string.

Three common approaches — all O(n):

// 1. Built-ins (cleanest)
const reverse = (s) => [...s].reverse().join("");

// 2. Loop (most explicit)
function reverseLoop(s) {
  let out = "";
  for (let i = s.length - 1; i >= 0; i--) out += s[i];
  return out;
}

// 3. Recursive (cute, but blows stack for long inputs)
const reverseRec = (s) => s === "" ? "" : reverseRec(s.slice(1)) + s[0];

reverse("hello"); // "olleh"

Spread handles surrogate pairs (emoji) correctly, which s.split("") does not. For production, use the built-in form.

Question 2

How would you implement a deep clone of an object?

Preferred: the built-in structuredClone. For environments without it, write a recursive helper that handles arrays, objects, dates, and cycles.

// Modern one-liner
const copy = structuredClone(obj);

// Polyfill
function deepClone(v, seen = new WeakMap()) {
  if (v === null || typeof v !== "object") return v;
  if (v instanceof Date)   return new Date(v);
  if (v instanceof RegExp) return new RegExp(v);
  if (seen.has(v))         return seen.get(v);   // cycle
  const out = Array.isArray(v) ? [] : {};
  seen.set(v, out);
  for (const k of Reflect.ownKeys(v)) out[k] = deepClone(v[k], seen);
  return out;
}

JSON.parse(JSON.stringify(o)) is short but drops functions, undefined, Maps, Sets, and corrupts Dates — avoid for real data.

Question 3

Write a function to flatten a nested array.

Two clean approaches:

// 1. Built-in — flatten to any depth
const flat = (arr) => arr.flat(Infinity);

// 2. Recursive — works in old environments
function flatten(arr) {
  const out = [];
  for (const item of arr) {
    if (Array.isArray(item)) out.push(...flatten(item));
    else out.push(item);
  }
  return out;
}

// 3. Iterative with stack — safe for very deep arrays
function flattenIter(arr) {
  const stack = [...arr], out = [];
  while (stack.length) {
    const next = stack.pop();
    if (Array.isArray(next)) stack.push(...next);
    else out.unshift(next);
  }
  return out;
}

flatten([1, [2, [3, [4]]]]); // [1,2,3,4]

Use flat(n) if you know the depth; flat(Infinity) for arbitrary nesting.

Question 4

Create a function that removes duplicates from an array.

A Set handles primitives in one line. For object duplicates, compare by a key.

// Primitives — O(n)
const unique = [...new Set(arr)];

// By a key (e.g., id)
function uniqueBy(arr, key) {
  const seen = new Set();
  return arr.filter(item => {
    const k = typeof key === "function" ? key(item) : item[key];
    if (seen.has(k)) return false;
    seen.add(k);
    return true;
  });
}

// Deep equality (slow — O(n²) JSON compare)
const uniqueDeep = (arr) =>
  arr.filter((v, i, a) => a.findIndex(x => JSON.stringify(x) === JSON.stringify(v)) === i);

unique([1,2,2,3]);              // [1,2,3]
uniqueBy(users, "id");          // dedupe by id
uniqueBy(users, u => u.email);  // dedupe by derived key
Question 5

Implement a function to check if a string is a palindrome.

Normalize first, then use a two-pointer scan or reverse-compare.

function isPalindrome(s) {
  const clean = s.toLowerCase().replace(/[^a-z0-9]/g, "");
  let i = 0, j = clean.length - 1;
  while (i < j) {
    if (clean[i] !== clean[j]) return false;
    i++; j--;
  }
  return true;
}

// Shorter (less memory-efficient)
const isPalindrome2 = (s) => {
  const c = s.toLowerCase().replace(/[^a-z0-9]/g, "");
  return c === [...c].reverse().join("");
};

isPalindrome("A man, a plan, a canal: Panama"); // true
isPalindrome("race a car");                     // false

Two-pointer is O(n) time, O(1) extra space. The reverse-compare version is O(n) time but O(n) space.

Question 6

Write a function to find the longest word in a string.

Split on whitespace and track the current max.

function longestWord(s) {
  return s.split(/\s+/)
          .reduce((best, w) => w.length > best.length ? w : best, "");
}

// Manual loop — slightly faster, no intermediate array
function longestWordLoop(s) {
  let best = "", cur = "";
  for (const ch of s + " ") {
    if (/\s/.test(ch)) {
      if (cur.length > best.length) best = cur;
      cur = "";
    } else cur += ch;
  }
  return best;
}

longestWord("The quick brown fox"); // "quick"

If tied, reduce keeps the first match. Strip punctuation beforehand (s.replace(/[^\w\s]/g, "")) if "quick," shouldn't count the comma.

Question 7

How would you merge two sorted arrays into one sorted array?

Count letter frequencies, or sort both strings — whichever is cleaner for the size of input.

// O(n) — frequency map
function isAnagram(a, b) {
  if (a.length !== b.length) return false;
  const count = {};
  for (const ch of a) count[ch] = (count[ch] || 0) + 1;
  for (const ch of b) {
    if (!count[ch]) return false;
    count[ch]--;
  }
  return true;
}

// O(n log n) — sort
const isAnagram2 = (a, b) =>
  a.length === b.length &&
  [...a].sort().join("") === [...b].sort().join("");

isAnagram("listen", "silent"); // true

Normalize case and whitespace first (a.toLowerCase().replace(/\s/g, "")) if inputs are phrases.

Question 8

Write a function to check if two strings are anagrams.

Classic closure + setTimeout.

function debounce(fn, wait) {
  let timer;
  return function (...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), wait);
  };
}

// With immediate option (fire on leading edge)
function debounce(fn, wait, immediate = false) {
  let timer;
  return function (...args) {
    const callNow = immediate && !timer;
    clearTimeout(timer);
    timer = setTimeout(() => {
      timer = null;
      if (!immediate) fn.apply(this, args);
    }, wait);
    if (callNow) fn.apply(this, args);
  };
}

const onSearch = debounce(q => fetch("/s?q=" + q), 300);
input.addEventListener("input", e => onSearch(e.target.value));

Lodash's _.debounce adds cancel/flush and leading+trailing edges.

Question 9

Implement a simple debounce function.

Three ways — recursion is clearest but limited by stack depth.

// Iterative — safest for large n
function factorial(n) {
  if (n < 0) throw new Error("negative");
  let r = 1n;                                  // BigInt for big n
  for (let i = 2n; i <= BigInt(n); i++) r *= i;
  return r;
}

// Recursive
const fact = (n) => n <= 1 ? 1 : n * fact(n - 1);

// Reduce
const factR = (n) => Array.from({length: n}, (_, i) => i + 1)
                         .reduce((a, b) => a * b, 1);

factorial(20);   // 2432902008176640000n
factorial(100);  // 9332621544394...

20! already exceeds Number.MAX_SAFE_INTEGER. Use BigInt for n > 20.

Question 10

Write a function to find the factorial of a number.

Classic backtracking: swap in place, recurse, swap back.

function permutations(s) {
  const chars = [...s];
  const out = [];
  function backtrack(start) {
    if (start === chars.length) {
      out.push(chars.join(""));
      return;
    }
    for (let i = start; i < chars.length; i++) {
      [chars[start], chars[i]] = [chars[i], chars[start]];
      backtrack(start + 1);
      [chars[start], chars[i]] = [chars[i], chars[start]];
    }
  }
  backtrack(0);
  return out;
}

// Dedupe if input has repeats
const uniquePerms = (s) => [...new Set(permutations(s))];

permutations("abc");
// ["abc","acb","bac","bca","cba","cab"]

n! permutations means n=10 gives 3.6M results — watch for memory blowups.

Question 11

Create a function to generate all permutations of a string.

Set makes this O(n+m).

function intersect(a, b) {
  const setA = new Set(a);
  return [...new Set(b.filter(x => setA.has(x)))];  // unique
}

// Preserve order from `a`, allow duplicates
function intersectWithDups(a, b) {
  const countB = new Map();
  for (const x of b) countB.set(x, (countB.get(x) || 0) + 1);
  const out = [];
  for (const x of a) {
    if (countB.get(x) > 0) {
      out.push(x);
      countB.set(x, countB.get(x) - 1);
    }
  }
  return out;
}

intersect([1,2,2,3], [2,3,4]); // [2,3]

Naive nested-loop approach is O(n·m) — avoid on large arrays.

Question 12

Implement a function to find the intersection of two arrays.

Check for own enumerable keys.

function isEmpty(obj) {
  if (obj == null) return true;
  return Object.keys(obj).length === 0 &&
         obj.constructor === Object;
}

// More robust — also handles Map, Set, arrays
function isEmptyAny(v) {
  if (v == null) return true;
  if (Array.isArray(v) || typeof v === "string") return v.length === 0;
  if (v instanceof Map || v instanceof Set) return v.size === 0;
  if (typeof v === "object") return Object.keys(v).length === 0;
  return false;
}

isEmpty({});         // true
isEmpty({a: 1});     // false
isEmpty(new Date()); // false — constructor check

Object.keys ignores inherited and non-enumerable properties — usually what you want.

Question 13

Write a function to check if an object is empty.

Wrap the function; cache results by serialized arguments.

function memoize(fn) {
  const cache = new Map();
  return function (...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) return cache.get(key);
    const result = fn.apply(this, args);
    cache.set(key, result);
    return result;
  };
}

// Classic example: exponential → linear
const fib = memoize(function (n) {
  return n < 2 ? n : fib(n - 1) + fib(n - 2);
});
fib(40); // fast

Caveats: JSON.stringify can't key by non-serializable args (functions, Symbols). For single-arg, use args[0] directly. Consider a WeakMap keyed by object args to avoid leaks.

Question 14

How would you create a memoized version of a function?

Divide-and-conquer on a sorted array — O(log n).

function binarySearch(arr, target) {
  let lo = 0, hi = arr.length - 1;
  while (lo <= hi) {
    const mid = (lo + hi) >> 1;          // fast ÷2 floor
    if (arr[mid] === target) return mid;
    if (arr[mid] < target) lo = mid + 1;
    else hi = mid - 1;
  }
  return -1;
}

// Recursive variant
function bsRec(arr, t, lo = 0, hi = arr.length - 1) {
  if (lo > hi) return -1;
  const mid = (lo + hi) >> 1;
  if (arr[mid] === t) return mid;
  return arr[mid] < t ? bsRec(arr, t, mid+1, hi) : bsRec(arr, t, lo, mid-1);
}

binarySearch([1,3,5,7,9,11], 7); // 3

Array must be sorted. Beware overflow for huge arrays: use lo + Math.floor((hi-lo)/2).

Question 15

const countVowels = (s) =>

// Manual loop — no regex
function countVowelsLoop(s) {

Question 16

Write a function to count the number of vowels in a string.

Use Number.prototype.toString(2), or build it manually.

// Built-in
const toBin = (n) => n.toString(2);

// Manual (positive integers)
function toBinary(n) {
  if (n === 0) return "0";
  let out = "";
  while (n > 0) {
    out = (n & 1) + out;
    n = n >>> 1;
  }
  return out;
}

// Negative — two's complement, 32-bit
const toBin32 = (n) => (n >>> 0).toString(2).padStart(32, "0");

toBin(10);      // "1010"
toBin(-5);      // "-101"  — sign + magnitude
toBin32(-5);    // "11111111111111111111111111111011"

For fractional numbers you'd need a different algorithm — rarely needed.

Question 17

Create a function to convert a number to its binary representation.

Classic but inefficient without memoization — O(2ⁿ). Memoize for O(n).

// Pure recursive — exponential
function fib(n) {
  if (n < 2) return n;
  return fib(n - 1) + fib(n - 2);
}

// Memoized
const fibMemo = (function () {
  const cache = [0, 1];
  return function f(n) {
    if (cache[n] !== undefined) return cache[n];
    return cache[n] = f(n - 1) + f(n - 2);
  };
})();

// Iterative — O(n) time, O(1) space
function fibIter(n) {
  let a = 0, b = 1;
  for (let i = 0; i < n; i++) [a, b] = [b, a + b];
  return a;
}

fibIter(50); // 12586269025

Use BigInt for n > 78 to avoid precision loss.

Question 18

Write a function to find the nth Fibonacci number using recursion.

Check divisibility up to √n.

function isPrime(n) {
  if (n < 2)        return false;
  if (n < 4)        return true;            // 2, 3
  if (n % 2 === 0)  return false;
  for (let i = 3; i * i <= n; i += 2) {
    if (n % i === 0) return false;
  }
  return true;
}

// 6k ± 1 optimization — skip obvious composites
function isPrime6k(n) {
  if (n < 2) return false;
  if (n < 4) return true;
  if (n % 2 === 0 || n % 3 === 0) return false;
  for (let i = 5; i * i <= n; i += 6) {
    if (n % i === 0 || n % (i + 2) === 0) return false;
  }
  return true;
}

isPrime(97);   // true
isPrime(100);  // false

For generating many primes, the Sieve of Eratosthenes is faster.

Question 19

Implement a function to check if a number is prime.

Recursion handles arbitrary nesting.

function deepSum(arr) {
  let total = 0;
  for (const item of arr) {
    total += Array.isArray(item) ? deepSum(item) : item;
  }
  return total;
}

// Alternative — flatten + reduce
const deepSum2 = (arr) =>
  arr.flat(Infinity).reduce((a, b) => a + b, 0);

deepSum([1, [2, [3, [4, 5]]]]); // 15

Recursive version is slightly faster (no intermediate array), but the flat + reduce version is more readable. Add typeof item === "number" guard if array may contain mixed types.

Question 20

Write a function to sum all the numbers in a nested array.

Uppercase the first letter of each word.

// Simple
const titleCase = (s) =>
  s.toLowerCase().replace(/\b\w/g, c => c.toUpperCase());

// Split/map — more control
const titleCase2 = (s) =>
  s.toLowerCase()
   .split(/\s+/)
   .map(w => w[0]?.toUpperCase() + w.slice(1))
   .join(" ");

// Skip small words (English)
const SMALL = new Set(["a","an","the","and","or","of","in","on"]);
function properTitle(s) {
  return s.toLowerCase().split(/\s+/).map((w, i) =>
    i === 0 || !SMALL.has(w) ? w[0].toUpperCase() + w.slice(1) : w
  ).join(" ");
}

titleCase("hello world"); // "Hello World"

Handle apostrophes carefully — \b matches before apostrophes so "it's" becomes "It'S". Use /(?:^|\s)\S/g to only touch the leading letter.

Question 21

Create a function to convert a string to title case.

Powers of two have exactly one bit set.

// Bitwise — O(1)
const isPow2 = (n) => n > 0 && (n & (n - 1)) === 0;

// Log-based (watch float precision)
const isPow2Log = (n) => {
  if (n <= 0) return false;
  const k = Math.log2(n);
  return Number.isInteger(k);
};

// Division loop
function isPow2Loop(n) {
  if (n <= 0) return false;
  while (n % 2 === 0) n /= 2;
  return n === 1;
}

isPow2(16);  // true
isPow2(18);  // false
isPow2(1);   // true — 2⁰

n & (n-1) clears the lowest set bit. If that yields zero, there was only one set bit — a pure power of two.

Question 22

Recursive structural comparison with cycle protection.

function deepEqual(a, b, seen = new WeakMap()) {
  if (Object.is(a, b)) return true;
  if (typeof a !== "object" || typeof b !== "object" ||
      a === null || b === null) return false;

  if (a.constructor !== b.constructor) return false;
  if (a instanceof Date)   return a.getTime() === b.getTime();
  if (a instanceof RegExp) return a.toString() === b.toString();

  if (seen.get(a) === b) return true;
  seen.set(a, b);

  const keysA = Reflect.ownKeys(a);
  const keysB = Reflect.ownKeys(b);
  if (keysA.length !== keysB.length) return false;
  for (const k of keysA) {

    if (!deepEqual(a[k], b[k], seen)) return false;
  }
  return true;
}

function sqrt(n, eps = 1e-10) {
  if (n < 0) return NaN;
  if (n === 0) return 0;
  let x = n;
  while (Math.abs(x * x - n) > eps) {

// Binary search variant (integer sqrt)
function isqrt(n) {
  if (n < 2) return n;
  let lo = 1, hi = n;
  while (lo <= hi) {
    const mid = (lo + hi) >> 1;
    if (mid * mid === n)      return mid;
    else if (mid * mid < n)   lo = mid + 1;
    else                       hi = mid - 1;
  }
  return hi;                             // floor(sqrt(n))
}

function kthLargest(nums, k) {
  const target = nums.length - k;          // index if sorted ascending
  const arr = [...nums];

  function select(lo, hi) {
    if (lo === hi) return arr[lo];
    const pivot = arr[hi];
    let i = lo;
    for (let j = lo; j < hi; j++) {
      if (arr[j] < pivot) {
        [arr[i], arr[j]] = [arr[j], arr[i]];
        i++;

    }

    if (i === target)  return arr[i];
    if (i < target)    return select(i + 1, hi);
    else                return select(lo, i - 1);
  }
  return select(0, arr.length - 1);
}

function editDistance(a, b) {
  const m = a.length, n = b.length;
  const dp = Array.from({length: m+1}, () => new Array(n+1).fill(0));

  for (let i = 0; i <= m; i++) dp[i][0] = i;

      if (a[i-1] === b[j-1]) {

      } else {

          dp[i-1][j],       // delete
          dp[i][j-1],       // insert
          dp[i-1][j-1]      // replace
        );
      }
    }
  }
  return dp[m][n];
}

function numDistinctIslands(grid) {
  const rows = grid.length, cols = grid[0].length;
  const shapes = new Set();

  const dfs = (r, c, r0, c0, path) => {
    if (r < 0 || r >= rows || c < 0 || c >= cols || grid[r][c] !== 1) return;
    grid[r][c] = 0;
    path.push((r - r0) + "," + (c - c0));
    dfs(r+1, c, r0, c0, path);
    dfs(r-1, c, r0, c0, path);
    dfs(r, c+1, r0, c0, path);
    dfs(r, c-1, r0, c0, path);
  };

  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {

    }
  }

}

function mergeStones(stones, k) {
  const n = stones.length;
  if ((n - 1) % (k - 1) !== 0) return -1;        // must leave exactly 1

  const prefix = [0];
  for (let i = 0; i < n; i++) prefix.push(prefix[i] + stones[i]);

  const dp = Array.from({length: n}, () => new Array(n).fill(0));

  for (let len = k; len <= n; len++) {
    for (let i = 0; i + len <= n; i++) {
      const j = i + len - 1;
      dp[i][j] = Infinity;
      for (let mid = i; mid < j; mid += k - 1) {
        dp[i][j] = Math.min(dp[i][j], dp[i][mid] + dp[mid+1][j]);
      }
      if ((len - 1) % (k - 1) === 0) dp[i][j] += prefix[j+1] - prefix[i];
    }
  }
  return dp[0][n-1];
}

Tricky DP. The mid stride of k-1 ensures we only split at merge-able boundaries.

function largestDivisibleSubset(nums) {
  if (!nums.length) return [];
  nums.sort((a, b) => a - b);
  const n = nums.length;

function maxPoints(points) {
  const n = points.length;
  if (n < 2) return n;

  const gcd = (a, b) => b === 0 ? a : gcd(b, a % b);

  let best = 0;
  for (let i = 0; i < n; i++) {
    const slopes = new Map();
    let duplicates = 1;
    for (let j = 0; j < n; j++) {
      if (i === j) continue;
      let dx = points[j][0] - points[i][0];
      let dy = points[j][1] - points[i][1];
      if (dx === 0 && dy === 0) { duplicates++; continue; }
      const g = gcd(Math.abs(dx), Math.abs(dy)) || 1;
      dx /= g; dy /= g;
      if (dx < 0) { dx = -dx; dy = -dy; }       // normalize direction
      const key = dx + "," + dy;
      slopes.set(key, (slopes.get(key) || 0) + 1);
    }
    let localBest = 0;
    for (const c of slopes.values()) localBest = Math.max(localBest, c);
    best = Math.max(best, localBest + duplicates);
  }
  return best;
}

Avoid floating-point slopes — they introduce rounding errors. O(n²).

function countComponents(n, edges) {
  const parent = Array.from({length: n}, (_, i) => i);

  const find = (x) => {
    while (parent[x] !== x) {
      parent[x] = parent[parent[x]];      // path compression
      x = parent[x];
    }
    return x;
  };
  const union = (a, b) => {
    const ra = find(a), rb = find(b);
    if (ra !== rb) { parent[ra] = rb; return true; }
    return false;
  };

  let components = n;
  for (const [a, b] of edges) if (union(a, b)) components--;
  return components;
}

function isMatch(s, p) {
  let i = 0, j = 0;
  let starJ = -1, starI = -1;
  while (i < s.length) {
    if (j < p.length && (p[j] === s[i] || p[j] === "?")) {
      i++; j++;
    } else if (j < p.length && p[j] === "*") {
      starJ = j;                 // record position of *
      starI = i;
      j++;                       // tentatively match zero chars
    } else if (starJ !== -1) {
      j = starJ + 1;             // backtrack: consume one more char with *
      i = ++starI;
    } else {
      return false;
    }
  }
  while (j < p.length && p[j] === "*") j++;
  return j === p.length;
}
